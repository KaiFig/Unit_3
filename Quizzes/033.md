```.py
def mystery(list1:list, list2:list)->list:
    outputlist = []
    for i in range(len(list1)):
        list1num = list1[i]
        for i in range(len(list2)):
            list2num = list2[i]
            if list1num == list2num:
                outputlist.append(list1num)

    return outputlist
```
